                    // Turn off default environment
                    viewer.setLightPreset(0);
                    log('Changed light preset');
                    
                    // Force redraw
                    viewer.impl.invalidate(true, true, true);

                }} else {{
                    log('Scene not available');
                }}

                log('Error applying skybox: ' + err.message);

        for (let i = 0; i < 6; i++) {{
            const url = urls[i];

            const onLoad = (image) => {{
                images[i] = image;

                if (loadedCount === 6) {{

                }}
            }};
            
            const onError = () => {{

            }};
